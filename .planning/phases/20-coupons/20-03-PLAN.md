       @Transactional
       public OrderDto createOrderFromCommand(String userId, CreateOrderCommand command) {
         if (userId == null || userId.isBlank()) {
           throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Missing X-User-Id session header");
         }

         validateStockOrThrow(command.items());

         BigDecimal subtotal = command.items().stream()
             .map(item -> item.unitPrice().multiply(BigDecimal.valueOf(item.quantity())))
             .reduce(BigDecimal.ZERO, BigDecimal::add);

         // === COUPON STEP (D-08, D-12) ===
         BigDecimal discountAmount = BigDecimal.ZERO;
         String appliedCouponCode = null;
         if (command.couponCode() != null && !command.couponCode().isBlank()) {
           // D-09: re-fetch by code; D-10: server compute discount từ subtotal
           String orderId = UUID.randomUUID().toString();   // ⚠ KHÔNG — orderId tạo bởi OrderEntity.create

           // Atomic redeem TRƯỚC khi tạo OrderEntity (vì insert redemption cần orderId, ta sẽ pre-allocate)
           // Pattern: tạo OrderEntity với UUID trước, lấy id, gọi atomicRedeem với id đó, rồi save.
           OrderEntity placeholder = OrderEntity.create(userId, BigDecimal.ZERO, "PENDING", command.note());
           CouponEntity coupon = couponRedemptionService.atomicRedeem(
               command.couponCode(), userId, placeholder.id());
           discountAmount = CouponPreviewService.computeDiscount(coupon, subtotal);
           appliedCouponCode = coupon.code();

           // Final total = subtotal - discount
           BigDecimal totalWithDiscount = subtotal.subtract(discountAmount);

           // Hydrate placeholder với final values
           String shippingAddressJson = serializeShipping(command.shippingAddress());
           placeholder.setShippingAddress(shippingAddressJson);
           placeholder.setPaymentMethod(command.paymentMethod());
           placeholder.setDiscountAmount(discountAmount);
           placeholder.setCouponCode(appliedCouponCode);
           // total cần update — dùng update() existing nếu signature cho phép, hoặc reflective set
           // Đơn giản nhất: thêm setTotal(BigDecimal) vào OrderEntity nếu chưa có; hoặc dùng update(...)
           placeholder.update(userId, totalWithDiscount, "PENDING", command.note());
           // Re-set fields update() vừa overwrite (do update reset note + status):
           // (Đảm bảo update signature giữ shippingAddress/paymentMethod nếu update không touch các field đó)

           for (OrderItemRequest itemReq : command.items()) {
             OrderItemEntity item = OrderItemEntity.create(placeholder, itemReq.productId(),
                 itemReq.productName(), itemReq.quantity(), itemReq.unitPrice());
             placeholder.addItem(item);
           }

           OrderEntity saved = orderRepository.save(placeholder);
           deductStockAfterPersist(command.items());
           return OrderMapper.toDto(saved);
         }

         // === NO COUPON PATH (existing logic) ===
         BigDecimal totalAmount = subtotal;
         String shippingAddressJson = serializeShipping(command.shippingAddress());

         OrderEntity order = OrderEntity.create(userId, totalAmount, "PENDING", command.note());
         order.setShippingAddress(shippingAddressJson);
         order.setPaymentMethod(command.paymentMethod());
         // discountAmount mặc định BigDecimal.ZERO từ field default; couponCode null
         for (OrderItemRequest itemReq : command.items()) {
           OrderItemEntity item = OrderItemEntity.create(order, itemReq.productId(),
               itemReq.productName(), itemReq.quantity(), itemReq.unitPrice());
           order.addItem(item);
         }
         OrderEntity saved = orderRepository.save(order);
         deductStockAfterPersist(command.items());
         return OrderMapper.toDto(saved);
       }

       private String serializeShipping(ShippingAddressRequest addr) {

           return objectMapper.writeValueAsString(addr);
         } catch (Exception e) {
           throw new ResponseStatusException(HttpStatus.BAD_REQUEST, "Invalid shippingAddress format");
         }
