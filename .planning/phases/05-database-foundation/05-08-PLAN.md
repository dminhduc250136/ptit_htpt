    ```bash
    GATEWAY_PORT=8080  # verify từ docker-compose.yml api-gateway block

    echo "$GW_RESPONSE" | python -c "import json,sys; d=json.load(sys.stdin); items=d if isinstance(d,list) else d.get('content',d.get('items',[])); assert len(items)==10, f'Expected 10 products, got {len(items)}'; print('Gateway returns', len(items), 'products')"
    ```

    **(b) DB ground truth match:**
    ```bash
    DB_NAMES=$(docker compose exec -T postgres psql -U tmdt -d tmdt -t -c "SELECT name FROM product_svc.products WHERE deleted=false ORDER BY id;" | sort | tr -d ' \r' | grep -v '^$')
    GW_NAMES=$(echo "$GW_RESPONSE" | python -c "import json,sys; d=json.load(sys.stdin); items=d if isinstance(d,list) else d.get('content',d.get('items',[])); print('\n'.join(sorted(p['name'].replace(' ','') for p in items)))")
    diff <(echo "$DB_NAMES") <(echo "$GW_NAMES") && echo "DB ground truth MATCH gateway response"
    ```

    ```bash
    echo "$GW_RESPONSE" | python -c "import json,sys; d=json.load(sys.stdin); items=d if isinstance(d,list) else d.get('content',d.get('items',[])); assert all('deleted' not in p for p in items), 'deleted field leaked!'; print('No deleted field leak')"
    ```

    ```bash
    mkdir -p .planning/phases/05-database-foundation/baseline
    for entry in user:8081 product:8082 order:8083 payment:8084 inventory:8085; do
      svc="${entry%%:*}"
      port="${entry##*:}"
      curl -s "http://localhost:${port}/v3/api-docs" \
        | python -c "import json,sys;json.dump(json.load(sys.stdin),sys.stdout,sort_keys=True,indent=2)" \
        > ".planning/phases/05-database-foundation/baseline/openapi-${svc}-service-post.json"
    done

    DIFF_FAILED=0
    for svc in user product order payment inventory; do
      if ! diff -q ".planning/phases/05-database-foundation/baseline/openapi-${svc}-service.json" ".planning/phases/05-database-foundation/baseline/openapi-${svc}-service-post.json"; then

        diff ".planning/phases/05-database-foundation/baseline/openapi-${svc}-service.json" ".planning/phases/05-database-foundation/baseline/openapi-${svc}-service-post.json" | head -50

      fi
    done
    test "$DIFF_FAILED" -eq 0 && echo "All 5 services OpenAPI shape unchanged"
    ```
