    **`mockChatStream.ts`** — Playwright `route.fulfill` returning a `text/event-stream` body composed of newline-delimited JSON events, mimicking the real route shape:
    ```typescript
    import type { Page } from '@playwright/test';

    export interface MockChatEvent {
      type: 'delta' | 'done' | 'error';
      text?: string;
      sessionId?: number;
      error?: string;
    }

    /** Intercept POST /api/chat/stream and return a deterministic SSE response. */
    export async function mockChatStream(page: Page, events: MockChatEvent[]) {
      await page.route('**/api/chat/stream', async route => {
        if (route.request().method() !== 'POST') return route.continue();
        const body = events.map(e => JSON.stringify(e)).join('\n') + '\n';
        await route.fulfill({
          status: 200,
          headers: {
            'Content-Type': 'text/event-stream; charset=utf-8',
            'Cache-Control': 'no-cache, no-transform',
          },
          body,
        });
      });
    }

    /** Convenience: build delta events from a string, splitting on word boundaries. */
    export function buildDeltas(fullText: string, sessionId = 1): MockChatEvent[] {
      const tokens = fullText.match(/\S+\s*/g) ?? [fullText];
      const out: MockChatEvent[] = tokens.map(t => ({ type: 'delta', text: t }));
      out.push({ type: 'done', sessionId, usage: undefined } as MockChatEvent);
      return out;
    }
    ```

              body: JSON.stringify({ type: 'delta', text: 'OK rồi.' }) + '\n' +

            });
          }
        });
        await page.goto('/');
        await page.getByTestId('chat-fab').click();
        await page.getByLabel('Soạn tin nhắn').fill('test');
        await page.getByLabel('Soạn tin nhắn').press('Enter');

        await page.getByRole('button', { name: 'Thử lại' }).click();
        await expect(page.locator('[data-role="assistant"]').last()).toContainText('OK rồi');
      });

      test('xml escape: user message with </product_context> không leak system instructions', async ({ page }) => {
        // Sanity: assistant fixture should NOT echo system instructions even if user injects close-tag.
        // Real protection lives server-side (escapeXml). This test only verifies the UI doesn't render raw HTML.
        await mockChatStream(page, [
          { type: 'delta', text: 'Mình chỉ tư vấn về sản phẩm thôi.' },
          { type: 'done', sessionId: 1 },

        // User input shown as text — no system tag rendered as HTML

        await expect(userBubble.locator('system')).toHaveCount(0);
      });

      test('no key leak: no NEXT_PUBLIC_ANTHROPIC_* anywhere in source tree', async () => {
        // Static grep over src/ — fails if any file references NEXT_PUBLIC_ANTHROPIC
        const root = path.resolve(__dirname, '../src');
        const found: string[] = [];
        async function walk(dir: string) {
          const entries = await fs.readdir(dir, { withFileTypes: true });
          for (const ent of entries) {
            const p = path.join(dir, ent.name);
            if (ent.isDirectory()) await walk(p);
            else if (/\.(ts|tsx|js|jsx|mjs|cjs|env|json)$/.test(ent.name)) {
              const text = await fs.readFile(p, 'utf-8');
              if (text.includes('NEXT_PUBLIC_ANTHROPIC')) found.push(p);
            }
          }
        }
        await walk(root);
        expect(found, `NEXT_PUBLIC_ANTHROPIC found in: ${found.join(', ')}`).toEqual([]);
      });
    });
